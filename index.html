create-react-tsx

A CLI tool to quickly generate beautiful React components with TypeScript support, straight from your terminal.

Install

npm install create-react-tsx
npx create-react-tsx <component-name>

Example

Create a Button component:

npx create-react-tsx Button

Generated Component

import React, { useState } from 'react';

interface ButtonProps {}

const Button: React.FC<ButtonProps> = () => {
  const [count, setCount] = useState(0);

  const increment = () => setCount(prev => prev + 1);
  const decrement = () => setCount(prev => prev - 1);

  return (
    <div className="p-8 flex flex-col items-center space-y-4">
      <div className="text-2xl font-bold">Counter App</div>
      <div className="text-4xl font-bold text-blue-500">{count}</div>
      <div className="flex space-x-4">
        <button onClick={increment} className="bg-blue-500 hover:bg-blue-700 text-white py-2 px-4 rounded">
          Increment
        </button>
        <button onClick={decrement} className="bg-red-500 hover:bg-red-700 text-white py-2 px-4 rounded" disabled={count === 0}>
          Decrement
        </button>
      </div>
    </div>
  );
};

export default Button;
